<script>
  // Register User
  function register() {
    const user = document.getElementById("regUser").value;
    const pass = document.getElementById("regPass").value;
    if (user && pass) {
      localStorage.setItem("user", user);
      localStorage.setItem("pass", pass);
      alert("Registration Successful!");
      document.getElementById("registerPage").style.display = "none";
      document.getElementById("loginPage").style.display = "block";
    } else {
      alert("Please fill all fields.");
    }
  }

  // Login User
  function login() {
    const user = document.getElementById("logUser").value;
    const pass = document.getElementById("logPass").value;
    if (user === localStorage.getItem("user") && pass === localStorage.getItem("pass")) {
      alert("Login Successful!");
      document.getElementById("loginPage").style.display = "none";
      document.getElementById("bookingPage").style.display = "block";
    } else {
      alert("Invalid credentials.");
    }
  }

  // Search Bus and Show Seats
  function searchBus() {
    const from = document.getElementById("from").value;
    const to = document.getElementById("to").value;
    if (from && to) {
      document.getElementById("seatSection").style.display = "block";
      const seatContainer = document.getElementById("seatContainer");
      seatContainer.innerHTML = "";
      for (let i = 1; i <= 16; i++) {
        const seat = document.createElement("div");
        seat.classList.add("seat");
        seat.innerHTML = `<img src="https://cdn-icons-png.flaticon.com/512/633/633652.png" alt="seat"> <br>Seat ${i}`;
        seat.onclick = function () {
          seat.classList.toggle("selected");
        };
        seatContainer.appendChild(seat);
      }
    } else {
      alert("Enter both locations.");
    }
  }

  // Confirm Seat Booking
  function confirmBooking() {
    const selectedSeats = document.querySelectorAll(".seat.selected");
    if (selectedSeats.length === 0) {
      alert("Please select at least one seat.");
      return;
    }
    let seats = [];
    selectedSeats.forEach(seat => seats.push(seat.innerText));
    alert("Seats booked: " + seats.join(", "));
    document.getElementById("trackingSection").style.display = "block";
  }

  // Simulated Bus Tracking
  function startTracking() {
    const stops = ["Starting Point", "Stop 1", "Stop 2", "Almost There", "Destination Reached"];
    let index = 0;
    const locationText = document.getElementById("busLocation");

    const interval = setInterval(() => {
      locationText.innerText = stops[index];
      index++;
      if (index >= stops.length) clearInterval(interval);
    }, 2000);
  }
</script>
